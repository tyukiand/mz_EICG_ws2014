\exercise{11.1}{Normal transformation}
\begin{itemize}
\item[a)]
\begin{align*}
	n'\cdot t'=(n')^T t&\stackrel{!}{=}n^Tt=n\cdot t\\
	n^TG^TMt&\stackrel{!}{=}n^Tt\\
	\Rightarrow G^TM&=\mathbbm{1}\\
	G^T&=M^{-1}\\
	\Rightarrow \hspace*{6mm} G\hspace*{2.5mm}&=\left(M^{-1} \right)^T 
\end{align*}
\end{itemize}
